161   def score_method(tally)
162     a, b, c = 0, 0, 0
163     tally.each do |cat, score|
164       case cat
165       when :assignment then a += score
166       when :branch     then b += score
167       else                  c += score
168       end
169     end
170     Math.sqrt(a*a + b*b + c*c)
171   end
