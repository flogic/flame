159   def score_method(tally)
160     a, b, c = 0, 0, 0
161     tally.each do |cat, score|
162       case cat
163       when :assignment then a += score
164       when :branch     then b += score
165       else                  c += score
166       end
167     end
168     Math.sqrt(a*a + b*b + c*c)
169   end
